import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split

def predict_readmission_risk(patient_data):
    """
    Predict patient readmission risk using Random Forest
    """
    # Feature engineering
    features = ['age', 'comorbidities', 'length_of_stay', 
                'previous_admissions', 'medication_count']
    
    X = patient_data[features]
    y = patient_data['readmitted_within_30d']
    
    # Split data
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
    
    # Train model
    model = RandomForestClassifier(n_estimators=100, random_state=42)
    model.fit(X_train, y_train)
    
    # Predictions
    predictions = model.predict_proba(X_test)[:, 1]
    
    return model, predictions

# Example usage

import pandas as pd
from statsmodels.tsa.arima.model import ARIMA
from sklearn.preprocessing import MinMaxScaler

def optimize_inventory(sales_data, lead_time=7):
    """
    Optimize inventory levels using time series forecasting
    """
    # Prepare data
    scaler = MinMaxScaler()
    scaled_data = scaler.fit_transform(sales_data.values.reshape(-1, 1))
    
    # Fit ARIMA model
    model = ARIMA(scaled_data, order=(1, 1, 1))
    model_fit = model.fit()
    
    # Forecast demand
    forecast = model_fit.forecast(steps=lead_time)
    forecast_actual = scaler.inverse_transform(forecast.reshape(-1, 1))
    
    # Calculate optimal stock levels
    safety_stock = np.std(forecast_actual) * 1.65  # 95% service level
    reorder_point = forecast_actual.mean() + safety_stock
    
    return {
        'forecast_demand': forecast_actual.flatten(),
        'reorder_point': reorder_point,
        'safety_stock': safety_stock
    }

import pandas as pd
import numpy as np
from sklearn.ensemble import GradientBoostingClassifier
from sklearn.metrics import classification_report, roc_auc_score
from sklearn.model_selection import cross_val_score

def predict_customer_churn(customer_data):
    """
    Predict customer churn using Gradient Boosting
    """
    # Feature engineering
    features = ['tenure', 'monthly_charges', 'total_charges',
                'contract_type', 'payment_method', 'tech_support']
    
    X = pd.get_dummies(customer_data[features])
    y = customer_data['churn']
    
    # Handle class imbalance
    from imblearn.over_sampling import SMOTE
    smote = SMOTE(random_state=42)
    X_resampled, y_resampled = smote.fit_resample(X, y)
    
    # Train model with cross-validation
    model = GradientBoostingClassifier(
        n_estimators=200,
        learning_rate=0.05,
        max_depth=4,
        random_state=42
    )
    
    # Evaluate with cross-validation
    cv_scores = cross_val_score(model, X_resampled, y_resampled, cv=5)
    model.fit(X_resampled, y_resampled)
    
    # Feature importance
    feature_importance = pd.DataFrame({
        'feature': X.columns,
        'importance': model.feature_importances_
    }).sort_values('importance', ascending=False)
    
    return model, feature_importance, cv_scores.mean()
